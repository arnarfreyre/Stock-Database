  Create New Endpoint for Available Stocks

  Add this new endpoint to src/backend/app.py (around line 250, after existing endpoints):

  @app.route('/api/available-stocks', methods=['GET'])
  def get_available_stocks():
      """Get list of all available stocks from CSV file."""
      try:
          import pandas as pd
          from pathlib import Path

          # Path to CSV file
          csv_path = Path(__file__).parent.parent / 'data' / 'nasdaq_screener.csv'

          if not csv_path.exists():
              return jsonify({'success': False, 'error': 'CSV file not found'}), 404

          # Read CSV file
          df = pd.read_csv(csv_path)

          # Convert to list of dictionaries
          available_stocks = []
          for _, row in df.iterrows():
              available_stocks.append({
                  'symbol': row['Symbol'],
                  'name': row['Name'],
                  'sector': row['Sector'] if pd.notna(row['Sector']) else 'N/A',
                  'industry': row['Industry'] if pd.notna(row['Industry']) else 'N/A'
              })

          return jsonify({
              'success': True,
              'available_stocks': available_stocks,
              'total_count': len(available_stocks)
          })

      except Exception as e:
          return jsonify({'success': False, 'error': str(e)}), 500

  Why this works:
  - Follows the same pattern as existing endpoints (/api/stocks)
  - Uses pandas to read CSV (already imported in your app)
  - Returns consistent JSON format with success flag
  - Handles errors gracefully
